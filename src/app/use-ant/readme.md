
**ant design组件库提供的组件模块如对于的名称下**

``` typescript
export { NzGridModule } ;//grid栅格
export { NzButtonModule } ;//按钮
export { NzAlertModule };// 弹出提示
export { NzBadgeModule };// 徽标数
export { NzCalendarModule } ;// 日历
export { NzCascaderModule };// 级联选择
export { NzCheckboxModule } ;// 多选框
export { NzDatePickerModule } ;// 日期选择框
export { NzFormModule } ;// 表单
export { NzInputModule }  ;// 输入框
export { NzInputNumberModule } ;// 数字输入框
export { NzMessageModule } ;// 全局提示
export { NzModalModule } ;// 对话框
export { NzNotificationModule } ;// 通知提醒框
export { NzPaginationModule } ;// 分页
export { NzPopconfirmModule } ;// 气泡确认框
export { NzPopoverModule } ;// 气泡卡片
export { NzRadioModule } ;// 单选框
export { NzRateModule } ;// 评分
export { NzSelectModule } ;// 选择器
export { NzSpinModule } ;// 加载中
export { NzProgressModule } ;// 进度条
export { NzSliderModule } ;// 滑动输入条
export { NzSwitchModule } ;// 开关
export { BgxTableModule } ;// 表格  如果想用功能更强大的表格，可以使用移植ngx-datatable的bg-datatable组件
export { NzTabsModule } ;// 标签页
export { NzTagModule } ;// 标签
export { NzTimePickerModule } ;//
export { NzStepsModule } ;// 步骤条
export { NzDropDownModule } ;// 下拉菜单
export { NzMenuModule } ;// 导航菜单
export { NzBreadCrumbModule } ;// 面包屑
export { NzLayoutModule } ;// 布局
export { NzCarouselModule } ;// 走马灯
export { NzCardModule } ;// 卡片
export { NzCollapseModule } ;// 折叠面板
export { NzTimelineModule } ;// 时间轴
export { NzToolTipModule } ;// 文字提示
export { NzBackTopModule } ;// 回到顶部
export { NzAffixModule } ;// 固钉
export { NzAnchorModule } ;// 锚点
export { NzAvatarModule } ;// 头像
export { NzNotificationService } ;// 通知提醒框
export { NzMessageService } ;// 全局提示 提供的服务 需要引入模块
export { NzModalService } ;// 对话框 提供的服务 需要引入模块

```
